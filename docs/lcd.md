# Lcd

Run with:
```bash
node eg/lcd.js
```


```javascript
var five = require("johnny-five"),
    lcd;

five.Board().on("ready", function() {


  lcd = new five.LCD({
    cols: 10,
    rows: 2,
    dots: "5x8"
  });


  this.repl.inject({
    lcd: lcd
  });



});

```
